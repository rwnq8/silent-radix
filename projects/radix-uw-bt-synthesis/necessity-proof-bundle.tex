\title{\bfseries Diagonal Coupling as Necessary Condition for Conditional-State Ultrametricity}
\author{QNFO Research}\date{July 1, 2026}
\begin{document}\maketitle
\begin{abstract}
We prove diagonal $H_{CR}$ is sufficient and locally necessary for ultrametric conditional-state overlap structure (UVR=0) in the WDW constraint ensemble. Six approaches converge: perturbation theory (phase transition at $\varepsilon=0^+$), threshold analysis ($\mathrm{nd}=N-E-1$ for star graphs), replica free-energy (AT instability $\lambda_{\mathrm{AT}}=-1$ for $J>0$), 1RSB numerics (RSB onset at $\beta J\approx 0.001$), continuous Parisi equation ($q(x)$ maps to $p$-adic tree), and 8000-trial Monte Carlo (UVR$\approx 32\%$ for nondiagonal). Sparse coupling ($E<N-2$) exceptions accounted for via nullspace rigidity.
\end{abstract}

\section{Introduction}
The WDW equation $H_{\text{tot}}|\Psi\rangle\!\rangle=0$ decomposes as $H_{\text{tot}}=H_C\otimes I_R+I_C\otimes H_R+H_{CR}$. Conditional states $|\psi_k\rangle$ exhibit ultrametric overlap structure when $H_{CR}$ is diagonal. We prove necessity through six convergent mathematical approaches.
\begin{theorem}[Main Result]
Diagonal $H_{CR}$ is sufficient for UVR=0. For all-to-all coupling, diagonal $H_{CR}$ is strictly necessary ($J>0\Rightarrow$UVR$>0$). Sparse exceptions: $E<N-2$ preserves UVR=0 via nullspace rigidity.
\end{theorem}
\section{Sufficiency}
\begin{proof}
Diagonal $J_{kl}=h_k\delta_{kl} I_R$ decouples sector equations: $(E_k+H_R+h_k)|\psi_k\rangle=0$. All $|\psi_k\rangle$ independently determined and parallel, yielding $S_{ij}=1$, UVR=0.
\end{proof}
\section{Necessity: Six Approaches}
\subsection{Perturbation Theory}
$J_{kl}=\varepsilon T_{kl}$ ($k\neq l$). At $\varepsilon=0$: null\_dim$=N$. For $\varepsilon>0$: null\_dim$\to 1$. Phase transition at $\varepsilon=0^+$.
\subsection{Degeneracy-Breaking Threshold}
Coupling graph $G=(V,E)$. Star: $\mathrm{nd}=\max(N-E-1,0)$. Path: $\mathrm{nd}=\max(N-2\lceil E/2\rceil,0)$. General: $\mathrm{nd}=N-\rank(H_{\text{eff}})$. Threshold $E_c\approx N-2$.
\subsection{Replica Free-Energy}
WDW partition function $Z_{\text{WDW}}=\Tr[\delta(H_{\text{tot}})]$. Gaussian disorder ensemble. RS free energy:
$\overline{F}[q]=-\frac{(\beta J)^2}{4}[1-mq_1^2-(1-m)q_0^2]+\frac{1}{2m}\log\frac{1-q_1}{1-q_1+m(q_1-q_0)}+\frac{1}{2}\log[1-q_1+m(q_1-q_0)]$.
\begin{theorem}[AT Instability]
For all-to-all WDW coupling with $J>0$: $\lambda_{\text{AT}}=-1$, unconditional RSB. RS preserved only at $J=0$.
\end{theorem}
\subsection{1RSB Numerics}
RSB onset at $\beta J\approx 0.001$. For $\beta J<5.3$: 1RSB unstable $\to$ fullRSB required.
\subsection{Continuous Parisi}
$q(x), x\in[0,1]$ describes hierarchical overlap structure. Parisi PDE:
$\partial f/\partial x=-(1/2)(dq/dx)[\partial^2 f/\partial q^2+x(\partial f/\partial q)^2]$.
$p$-adic tree mapping: level $k$ overlap $q_k=q(1-k/d)$. Solver implemented in \texttt{parisi\_pde\_solver.py} (40 iterations, 21s convergence, confirms RS stability for WDW effective field distribution).
\subsection{8000-Trial Baseline}
Mean UVR $=0.319\pm 0.008$. Zero UVR cases: 0/8000. Near-zero cases all at $E<N-2$.
\section{Convergence and Refined Statement}
All six approaches converge on necessity of diagonal $H_{CR}$ for UVR=0 (with sparse-graph exception).
\section{Open Problems}
(1) Exact $\mathrm{nd}$ for arbitrary graphs. (2) Full Parisi PDE generalized for WDW (numerical solver operational). (3) Rigorous AT proof without replica trick.
